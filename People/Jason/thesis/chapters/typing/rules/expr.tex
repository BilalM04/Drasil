\begin{enumerate}

    \item Literals:
        \newrule{\ty{x} & \ofTy{l}{Literal x}}
            {\ofTy{Lit[l]}{Expr x}}

    \item Associative Operations:
        \newrule{\ty{x} & \ofTy{op}{AssocOp x} & \ofTy{args}{[Expr x]}}
            {\ofTy{Assoc[op, args]}{Expr x}}

    \item Symbols:
        \newrule{\ty{x} & \ofTy{u}{UID}}
            {\ofTy{C[u]}{Expr x}}
          
    \item Function Call: \todo{discuss functions in general}
    
    \item Case:
        \newrule{\ty{x} & \ofTy{c}{Completeness} & \ofTy{ces}{[(Expr Bool, Expr x)]}}
            {\ofTy{Case[c, ces]}{Expr x}}
    
    \item Matrices:
        \newrule{\ty{x} & \ofTy{es}{[[Expr x]]}}
            {\ofTy{Matrix[es]}{Expr x}}

    \item Unary Operations:
        \newrule{\ty{x} & \ty{y} & \ofTy{op}{UnaryOp x y} & \ofTy{e}{Expr x}}
            {\ofTy{Unary[op, e]}{Expr y}}

    \item Binary Operations:
        \newrule{\ty{x} & \ty{y} & \ty{z} & \ofTy{op}{BinaryOp x y z} & \ofTy{l}{Expr x} & \ofTy{r}{Expr y}}
            {\ofTy{Binary[op, l, r]}{Expr z}}

    \item ``Big'' Operations:
        \newrule{\ty{x} & \ofTy{op}{AssocOp x} & \ofTy{dom}{DomainDesc Discrete (Expr x) (Expr x)}}
            {\ofTy{BigOp[op, dom]}{Expr x}}
    
    \item ``Is in interval'' operator:
        \newrule{\ty{x} & \ofTy{u}{UID} & \ofTy{itvl}{RealInterval (Expr x) (Expr x)}}
            {\ofTy{RealI[u, itvl]}{Expr x}}

\end{enumerate}
