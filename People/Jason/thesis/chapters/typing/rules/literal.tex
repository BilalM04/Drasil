\begin{enumerate}

    \item Integers:
        \newrule{\ofTy{i}{Integer}}
            {\ofTy{Integer[i]}{Literal Integer}}

    \item Strings (Text):
        \newrule{\ofTy{s}{String}}
            {\ofTy{Str[s]}{Literal String}}

    \item Real numbers:
        \newrule{\ofTy{d}{Double}}
            {\ofTy{Dbl[d]}{Literal Real}}

    \item Whole numbered reals (\(\bb{Z} \subset{} \bb{R}\)): 
        \newrule{\ofTy{d}{Integer}}
            {\ofTy{ExactDbl[d]}{Literal Real}}

    \item Percentages:
        \newrule{\ofTy{n}{Integer} & \ofTy{d}{Integer}}
            {\ofTy{Perc[n,d]}{Literal Real}}

\end{enumerate}
